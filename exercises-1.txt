1. Pseudocode Soal 1

SET massa WITH VALUE 600
SET percepatan WITH VALUE 2

SET resultangaya WITH CALCULATE OF massa x percepatan

DISPLAY resultangaya

2. Tahun Kabisat

SET tahun WITH VALUE 2019

IF tahun MOD 4 === 0 AND tahun MOD 100 !== 0 THEN
    DISPLAY "TAHUN KABISAT"
ELSE IF tahun MOD 4 === 0 AND tahun MOD 100 === 0 AND tahun MOD 400 === 0 THEN
    DISPLAY "TAHUN KABISAT
ELSE
    DISPLAY "BUKAN TAHUN KABISAT"
ENDIF

3. Laundry Day

SET pakaian WITH VALUE 1

WHILE pakaian <= 20
    pakaian INCREMENT
ENDWHILE

IF pakaian === 20 THEN
    DISPLAY "MESIN CUCI DINYALAKAN"
ELSE
    DISPLAY "JUMLAH PAKAIAN BELUM SAMPAI 20"
ENDIF

4. PERIKSA KUKU

SET iterasi WITH VALUE 1
SET jumlahsiswa WITH VALUE 40
SET siswa WITH VALUE 'Kuku Panjang'

WHILE iterasi <= jumlahsiswa

    IF siswa === 'Kuku Panjang' THEN
        DISPLAY 'MENDAPAT HUKUMAN DARI GURU'
    ELSE
        DISPLAY 'MENDAPAT PUJIAN DARI GURU'
    ENDIF
    
    iterasi INCREMENT

ENDWHILE
